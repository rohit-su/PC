#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

int main(int argc, char *argv[]) {
    if (argc != 2) return 1;
    int n = atoi(argv[1]);

    #pragma omp parallel for schedule(static,2)
    for (int i = 0; i < n; i++) {
        if (i % 2 == 0) {                // start of a 2-iteration chunk
            int end = (i + 1 < n) ? i + 1 : n - 1;
            printf("Thread %d: Iterations %d -- %d\n",
                   omp_get_thread_num(), i, end);
        }
    }
    return 0;
}
