#include <mpi.h>
#include <stdio.h>

int main(int argc, char **argv) {
    MPI_Init(&argc, &argv);

    int rank, data;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    if (rank == 0) data = 42;   // root sets the value

    MPI_Bcast(&data, 1, MPI_INT, 0, MPI_COMM_WORLD);

    printf("Process %d received %d\n", rank, data);

    MPI_Finalize();
    return 0;
}
